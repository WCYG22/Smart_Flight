`; recommendationsContainer.innerHTML = ''; } } function displayAlternatives(alternatives) { const container = document.getElementById('alternativesContainer'); let html = '

'✈️' : alt.type === 'different_airline' ? '🔄' : '⏰'; html += `
${reasonIcon} ${alt.reason}
${alt.flight_number} ${alt.risk_level}
${alt.origin} → ${alt.destination}
${depTime} - ${arrTime}
Reliability: ${alt.reliability_score}% • Risk: ${Math.round(alt.disruption_probability * 100)}%
`; }); container.innerHTML = html; } function displayRecommendations(recommendations) { const container = document.getElementById('recommendationsContainer'); let html = '
